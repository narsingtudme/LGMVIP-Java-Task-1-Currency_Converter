    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        
        if(FirstCountry.getSelectedIndex()==0 || secondCountry.getSelectedIndex()==0 || T1.getText().equals(""))
        {
            JOptionPane.showMessageDialog(null,
                    "you must select both countries and must input the amount", 
                    "Error message", JOptionPane.INFORMATION_MESSAGE);
            
                    return;
        
        }
        
        double amountToChange = Double.parseDouble(T1.getText());
        double amountInPounds= 0.0;
        
        double amountChanged= 0.0;
        
        switch (FirstCountry.getSelectedItem().toString())
        {
            case "USA": amountInPounds = amountToChange / USA_Dollar; break;
            case "Brazil": amountInPounds = amountToChange / Brazil_Real; break;
            case "India": amountInPounds = amountToChange / India_Rupee; break;
            case "Pakistan": amountInPounds = amountToChange / Pakistan_Rupee; break;
            case "Canada": amountInPounds = amountToChange / Canada_Dollar; break;
            case "Europe": amountInPounds = amountToChange / Europe_Dollar; break;
           // case "Indoneisa": amountInPounds = amountToChange / Indoneisa_Rupee; break;
            default:amountInPounds=0.0;
            
        }
        //amount is changed in pounds
        
        switch (secondCountry.getSelectedItem().toString())
        {
           case "USA": amountChanged = amountInPounds * USA_Dollar; break;
            case "Brazil": amountChanged = amountInPounds * Brazil_Real; break;
            case "India": amountChanged = amountInPounds * India_Rupee; break;
            case "Pakistan": amountChanged = amountInPounds * Pakistan_Rupee; break;
            case "Canada": amountChanged = amountInPounds * Canada_Dollar; break;
            case "Europe": amountChanged = amountInPounds * Europe_Dollar; break;
            //case "Indoneisa": amountChanged = amountInPounds * Indoneisa_Rupee; break;
            default:amountChanged = amountInPounds*0.0;
            
        }
        
        String value =String.format("%.2f",amountChanged);
        
        T2.setText(value);        
        
    }
